Sub Prueba()

 ' Obtener objeto Coleccion de documentos
 Dim invDocs As Documents
 Set invDocs = ThisApplication.Documents
 
 ' Iterar a traves del contenido de la Coleccion de documentos
 Dim i As Integer
 For i = 1 To invDocs.Count
 
    ' Obtener documento de la Coleccion de documentos
    Dim invDocument As Document
    Set invDocument = invDocs.Item(i)
    
    ' Mostrar el documento en consola
    Debug.Print invDocument.FullFileName

 Next

' Iterar con foreach
For Each invDocument In invDocs
	' Mostrar el documento en consola.
	Debug.Print invDocument.FullFileName
Next

End Sub
