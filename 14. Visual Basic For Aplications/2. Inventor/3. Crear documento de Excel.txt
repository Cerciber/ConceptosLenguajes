
Sub prueba()

    ' Crear objeto Excel
    Dim objExcel As Excel.Application
    Set objExcel = CreateObject("Excel.Application")
    
    ' Crear objeto libro de trabajo con la ruta especificada
    Dim objWorkbook As Workbook
    Set objWorkbook = Workbooks.Add
    
    ' Crear objeto hoja con el nombre especificado
    Dim objSheet As WorkSheet
    Set objSheet = objWorkbook.Sheets.Add
    
    ' Obtener objeto celda en la posicion especificada
    Dim objCell As Range
    Set objCell = objSheet.Range("A1")
    
    ' Cambiar el valor de una celda
    objCell = "Nuevo valor"
    
    ' Imprimir celda en colsola
    Debug.Print objCell.Value
    
    ' Guardar copia del libro en la ubicacion especificada
    objWorkbook.SaveCopyAs "C:\Users\cesar\Downloads\60507229-9-pallet_jack_yellow\prueba2.xlsx"
    
    ' Mostrar Excel al usuario
    objExcel.Application.Visible = True
    
End Sub
